CREATE DATABASE PRODUCTOS;

USE PRODUCTOS;

CREATE TABLE PRODUCTO(
	ID_PRODUTO INT AUTO_INCREMENT PRIMARY KEY,
    NOMBRE VARCHAR(50),
    NOSERIE VARCHAR(50),
    EXISTENCIA INT
);

CREATE PROCEDURE insertarProducto(
	IN NOMBRE_PRODUTO VARCHAR(50),
    IN NOSERIE_PRODUCTO VARCHAR(50),
    IN EXISTENCIA_PRODUCTO INT	
) INSERT INTO PRODUCTO(NOMBRE, NOSERIE, EXISTENCIA) VALUES (NOMBRE_PRODUTO, NOSERIE_PRODUCTO, EXISTENCIA_PRODUCTO);

SELECT * FROM PRODUCTO;
